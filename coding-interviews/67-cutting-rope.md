def cutRope(number):
    a = [0] * 60
    a[1] = 1
    a[2] = 2
    a[3] = 3
    for i in range(4, number + 1):
        max = 0
        for j in range(1, int(i // 2) + 1):
            a[i] = a[j] * a[i - j]
            if max < a[i]:
                max = a[i]
            a[i] = max
    return a[number]
