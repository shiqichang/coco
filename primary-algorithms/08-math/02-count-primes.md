```python
def count_primes(n):
    is_prime = [True] * n
    for i in range(2, int(math.sqrt(n)) + 1):
        if is_prime[i]:
            for j in range(i*i, n, i):
                is_prime[j] = False

    count = 0
    for i in range(2, n):
        if is_prime[i]:
            count += 1
    return count
```
